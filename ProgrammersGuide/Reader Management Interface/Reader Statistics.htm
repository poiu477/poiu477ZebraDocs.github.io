Reader Statistics

The method rm.getReaderStatistics can be used for getting the reader statistics for each antenna. The method rm.clearReaderStatistics can be used to clear the Reader Statistics.

try {

    short antennaID = 1;

    System.out.println("Reader Statistics for AntennaID " + antennaID);

 

    // Get Reader Statistics          

    ReaderStatistics stats = rm.getReaderStatistics(antennaID);

    System.out.println(" Identified Success Count       : " + stats.getIdentifiedSuccessCount());

    System.out.println(" Identified Failure Count       : " + stats.getIdentifiedFailureCount());

    System.out.println(" Read Success Count             : " + stats.getReadSuccessCount());

    System.out.println(" Read Failure Count             : " + stats.getReadFailureCount());

    System.out.println(" Write Success Count            : " + stats.getWriteSuccessCount());

    System.out.println(" Write Failure Count            : " + stats.getWriteFailureCount());

    System.out.println(" Kill Success Count             : " + stats.getKillSuccessCount());

    System.out.println(" Kill Failure Count             : " + stats.getKillFailureCount());

    System.out.println(" Lock Success Count             : " + stats.getLockSuccessCount());

    System.out.println(" Lock Failure Count             : " + stats.getLockFailureCount());

 

    // Clear Reader Statistics          

    rm.clearReaderStatistics();

 

} catch (OperationFailureException exception) {

    System.out.println("testReaderStats: GetStats failed " + exception.getVendorMessage());

} catch (InvalidUsageException exception){

}
